66  void update() {
67  if(!cursor) return;
68  std::size_t i = cursor;
69  T *tmp(nullptr);
70  while(i--) {
71  if(arr[i]->update()) continue;
72  if(i != --cursor) {
73  tmp = arr[i];
74  arr[i] = arr[cursor];
75  arr[cursor] = tmp;
76  }
77  }
78  }
